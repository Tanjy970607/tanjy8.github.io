import pandas as pd
import localization_api as loc

# Process translation & files
engine = loc.Translator(languages=["EN", "ZH", "MS"])

def process_job(client_file):
    df = pd.read_excel(client_file)
    clean_df = df.drop_duplicates()
    translated = engine.translate(clean_df)
    return translated.export_ppt(theme="Keynote")
